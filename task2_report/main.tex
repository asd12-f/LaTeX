\section*{Вариант 12}

Введем обозначения

\[
\begin{aligned}
2K_{sk} &= \frac{12\gamma\omega}{2\varepsilon^2\alpha} M_{sk}; \quad 
2K_{ck} = \frac{12\gamma\omega}{2\varepsilon^2\alpha} M_{ck}; \\
M_{ck} &= \frac{\rho\nu}{\delta_0\psi^2} \left[ \frac{2}{(2k-1)\pi} \right]^2 \frac{2\varepsilon^2\alpha}{\omega}; \quad 
M_{sk} = \frac{\rho\nu}{\delta_0\psi^2} \left[ \frac{1}{k\pi} \right]^2 \frac{2\varepsilon^2\alpha}{\omega}; \\
a_{1ck} &= \left(\frac{(2k-1)\pi}{2\ell}\right)^4 - [\rho_0h_0 + M_{ck}]\omega^2/D; \quad 
a_{1sk} = \left(\frac{k\pi}{\ell}\right)^4 - [\rho_0h_0 + M_{sk}]\omega^2/D; \\
a_{2ck} &= 2K_{ck}\omega/D; \quad 
a_{2sk} = 2K_{sk}\omega/D.
\end{aligned}
\]
